# all other cities' data
cities_stats = 
  homicide_summary_df %>% 
  nest(reg_data = -city_state) %>% 
  mutate(
    reg_line = map(.x = reg_data, ~glm(disposition ~ victim_age + victim_sex + victim_race, data = .x, family = binomial())),

  ) %>% 
  select(-reg_data) %>% 
  unnest(reg_line) %>% 
  filter(term == "victim_sexMale") %>% 
  mutate(
